# Python program to calculate the first 10,000 terms of the series
terms = 10000
total_sum = 0.0

for i in range(terms):
    # Calculate the current term: (1/(2*i + 1)) with alternating signs
    term = (-1)**i / (2*i + 1)
    total_sum += term

# Multiply the total sum by 4
total_sum *= 4

# Print the final result
print(total_sum)

Output: 3.1414926535900345
